 where (t1.SZ_PID0='320000000' AND t0.HZRQQ<to_timestamp('20160412','yyyymmdd') AND t0.HZRQZ>=to_timestamp('20160411','yyyymmdd') AND (t0.JYFW LIKE '%理财%' OR t0.JYFW LIKE '%投资%' OR t0.JYFW LIKE '%资产管理%')) and (t0.CANCEL_FLAG IS NULL) and (t0.COME_OUT IS NULL OR t0.COME_OUT='0')
